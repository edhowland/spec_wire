def conditionally_execute(cmd, dfile, *args)
  if (File.exists? args.last and !force?) or File.directory? args.last
    puts "exists " + dfile
  else
    puts "create #{dfile}"
    self.send cmd, *args unless dry_run?
  end
end

def die(msg, *args)
  puts msg + (args.empty? ? '' : ' |' + args.map {|e| e.inspect}.join('|'))
  exit
end
